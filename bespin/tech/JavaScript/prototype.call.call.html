# Function.prototype.call.call

function fn1() {
    console.log(111111111);
}

function fn2() {
    console.log(222222222);
}

fn1.call(fn2);

fn1.call.call(fn2);

prototype.call = ctx => {
    const context = ctx || window;
    context.fn = this;
    return ctx.fn();
}

再来看第一个

fn1.call(fn2);

call里面当前this指向fn1，执行fn就是执行this，context是fn2，fn中的this指向context，也就是改变了fn1的this

再来看第二个

fn1.call.call(fn2);

从右向左执行 fn1.call.call(fn2)， 遇到第一个call，call当前this指向fn1.call，context变成fn2，改变了fn1.call的this，指向fn2 执行fn1.call() 遇到第二个call，call当前this指向第一个call的context也就是fn2，执行this就是执行fn2，参数是空，context变成window

fn1.call(fn2); // 相当于 把fn1()挂载到fn2上，即fn2.fn1(this=fn2)() fn1.call.call === Function.prototype.call; Function.prototype.call.call(fn2); // 相当于把 Function.prototype.call挂载在fn2 上，即 // fn2.(Function.prototype.call(this=this))() // 即相当于fn2.call(); 验证：
